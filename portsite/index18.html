    const word = document.querySelector(".word")
const shadow = (e) =>{
    const {x , y} = e;
    const rect = word.getBoundingClientRect()
    // console.log(rect)
    const mouseX = (x - rect.left - rect.width/2) /10
    const mouseY = (y - rect.top - rect.height/2) /10

    gsap.to(".cursor", {duration: .2, left: e.pageX - 10, top: e.pageY - 10});

    word.style.textShadow = `${mouseX}px ${-mouseY}px 0px rgba(102,249,255,0.7),
                                ${-mouseX}px ${mouseY}px 0px rgba(16, 238, 145, 0.7),
                                ${mouseY}px ${-mouseX}px 0px rgba(255,255,73,0.7),
                                ${-mouseX}px ${mouseX}px 0px rgba(16, 238, 145, 0.7)`;
}

window.addEventListener("mousemove", shadow)
